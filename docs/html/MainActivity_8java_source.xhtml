453  private void resetAll(boolean keepRdmUnit) {
454  if (sharedPreferences.getBoolean("sound_reset", true)) {
455  playSound(R.raw.reset);
456  }
457  int revengeCount = 0;
458  if (keepRdmUnit) {
459  HashSet<Row> fullClear = new HashSet<>();
460  for (Row row : allRows) {
461  if (row.isRevenge()) {
462  revengeCount += row.getJaskierMoralBoosterRevengeCount()[2];
463  }
464  if (row.getCardCount() == 0 || row.isEpicOnly()) {
465  fullClear.add(row);
466  }
467  }
468  for (Row row : fullClear) {
469  allRows.remove(row);
470  }
471  int size = allRows.size();
472  while (size > 1) {
473  int randomItem = new Random().nextInt(size);
474  int i = 0;
475  for (Row row : allRows) {
476  if (i == randomItem) {
477  fullClear.add(row);
478  break;
479  }
480  i++;
481  }
482  for (Row row : fullClear) {
483  allRows.remove(row);
484  }
485  size = allRows.size();
486  }
487  for (Row row : allRows) {
488  row.clear(true);
489  Unit keepUnit = row.getAllUnits().get(0);
490  Toast.makeText(this, getString(R.string.alertDialog_factionreset_monster_toast_keep, keepUnit.toString(this, null)), Toast.LENGTH_LONG).show();
491  if (keepUnit.isRevenge()) {
492  revengeCount--;
493  }
494  }
495  if (allRows.size() == 0) {
496  Toast.makeText(this, R.string.alertDialog_factionreset_monster_toast_nokeep, Toast.LENGTH_LONG).show();
497  }
498  for (Row row : fullClear) {
499  row.clear(false);
500  allRows.add(row);
501  }
502  } else {
503  for (Row row : allRows) {
504  if (row.isRevenge()) {
505  revengeCount += row.getJaskierMoralBoosterRevengeCount()[2];
506  }
507  row.clear(false);
508  }
509  }
510  resetHorns();
511  resetWeather();
512  inflateRevengeDialog(revengeCount);
513  updateRows();
514  }
