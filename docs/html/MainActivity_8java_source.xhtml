191 private void initializeViewModel() {
192 int[] rowIds = {R.id.firstRow, R.id.secondRow, R.id.thirdRow};
193 for (int rowId = 0; rowId < rowIds.length; rowId++) {
194 RowType row = RowType.values()[rowId];
195 ConstraintLayout rowLayout = findViewById(rowIds[rowId]);
196
197 ImageView weather = rowLayout.findViewById(R.id.weatherView);
198 ImageView horn = rowLayout.findViewById(R.id.hornView);
199 ConstraintLayout cards = rowLayout.findViewById(R.id.cardView);
200
201 weather.setOnClickListener(v ->
202 disposables.add(gameBoardViewModel.onWeatherViewPressed(row).subscribe()));
203 horn.setOnClickListener(v ->
204 disposables.add(gameBoardViewModel.onHornViewPressed(row).subscribe()));
205
206 cards.setOnClickListener(v -> disposables.add(
207 GwentApplication.getRepository(this)
208 .flatMap(repository -> repository.countUnits(row))
209 .map(count -> count == 0)
210 .flatMap((Function<Boolean, Single<? extends OverlayDialog>>) rowEmpty ->
211 rowEmpty
212 ? Single.just(new AddCardDialog(MainActivity.this, row, gameBoardViewModel.getSoundManager()))
213 : ShowUnitsDialog.getDialog(MainActivity.this, row, gameBoardViewModel.getSoundManager()))
214 .subscribe(Dialog::show)
215 ));
216
217 final RowUiStateObserver observer = RowUiStateObserver.getObserver(row,
218 rowLayout.findViewById(R.id.pointView),
219 weather,
220 horn,
221 rowLayout.findViewById(R.id.cardCountView));
222 disposables.add(gameBoardViewModel.getRowUiState(row).subscribe(observer));
223 }
224
225 ImageButton reset = findViewById(R.id.resetButton);
226 ImageButton weather = findViewById(R.id.weatherButton);
227 ImageButton burn = findViewById(R.id.burnButton);
228
229 final MenuUiStateObserver observer = new MenuUiStateObserver(findViewById(R.id.overallPointView),
230 reset,
231 weather,
232 burn);
233 disposables.add(gameBoardViewModel.getMenuUiState().subscribe(observer));
234
235 reset.setOnClickListener(v ->
236 disposables.add(gameBoardViewModel.onResetButtonPressed(this).subscribe()));
237 weather.setOnClickListener(v ->
238 disposables.add(gameBoardViewModel.onWeatherButtonPressed().subscribe()));
239 burn.setOnClickListener(v ->
240 disposables.add(gameBoardViewModel.onBurnButtonPressed(MainActivity.this).subscribe()));
241 }

250 private void inflateFactionPopup() {
251 new ChangeFactionDialog(this, theme -> {
252 SharedPreferences preferences = PreferenceManager.getDefaultSharedPreferences(this);
253
254 if (preferences.getInt(THEME_PREFERENCE_KEY, THEME_SCOIATAEL) == theme) {
255 return;
256 }
257
258 boolean resetOnFactionSwitch = preferences.getBoolean(
259 getString(R.string.preference_key_faction_reset),
260 getResources().getBoolean(R.bool.faction_reset_preference_default)
261 );
262 if (resetOnFactionSwitch) {
263 disposables.add(gameBoardViewModel.onFactionSwitchReset(this).subscribe());
264 }
265 preferences.edit()
266 .putInt(THEME_PREFERENCE_KEY, theme)
267 .apply();
268 }).show();
269 }
