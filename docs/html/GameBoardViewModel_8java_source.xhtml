96  for (RowType row : RowType.values()) {
97  result.rowUiStates.put(row,
98  Flowable.combineLatest(repository.isWeatherFlowable(row),
99  repository.isHornFlowable(row),
100  repository.getUnitsFlowable(row),
101  (weather, horn, units) -> {
102  DamageCalculator calculator = DamageCalculatorUseCase.getDamageCalculator(weather, horn, units);
103  int damage = units.stream()
104  .map(unitEntity -> unitEntity.calculateDamage(calculator))
105  .reduce(0, Integer::sum);
106  return new RowUiState(damage, weather, horn, units.size());
107  })
108  .distinctUntilChanged()
109  .onBackpressureLatest()
110  .debounce(10, TimeUnit.MILLISECONDS)
111  .subscribeOn(Schedulers.io())
112  .observeOn(AndroidSchedulers.mainThread())
113  );
114  }

116  Flowable<MenuUiState> combinedRowStates = Flowable.combineLatest(result.rowUiStates.values(), (Object[] rowUiStates) -> {
117  int damage = 0;
118  boolean reset = false;
119  boolean weather = false;
120  for (Object state : rowUiStates) {
121  RowUiState rowUiState = (RowUiState) state;
122  damage += rowUiState.getDamage();
123  reset |= rowUiState.isHorn() || rowUiState.getUnits() != 0;
124  weather |= rowUiState.isWeather();
125  }
126  reset |= weather;
127  return new MenuUiState(damage, reset, weather, false);
128  }).distinctUntilChanged().onBackpressureLatest();
129
