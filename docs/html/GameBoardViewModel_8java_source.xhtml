97 @NonNull
98 public static GameBoardViewModel getModel(@NonNull ViewModelStoreOwner owner,
99 @NonNull UnitRepository repository,
100 @NonNull SoundManager soundManager) {
101 GameBoardViewModel result = new ViewModelProvider(owner, ViewModelProvider.Factory.from(INITIALIZER))
102 .get(GameBoardViewModel.class);
103
104 for (RowType row : RowType.values()) {
105 result.rowUiStates.put(row,
106 Flowable.combineLatest(repository.isWeatherFlowable(row),
107 repository.isHornFlowable(row),
108 repository.getUnitsFlowable(row),
109 (weather, horn, units) -> {
110 DamageCalculator calculator = DamageCalculatorUseCase.getDamageCalculator(weather, horn, units);
111 int damage = units.stream()
112 .map(unitEntity -> unitEntity.calculateDamage(calculator))
113 .reduce(0, Integer::sum);
114 return new RowUiState(damage, weather, horn, units.size());
115 })
116 .distinctUntilChanged()
117 .onBackpressureLatest()
118 .debounce(10, TimeUnit.MILLISECONDS)
119 .subscribeOn(Schedulers.io())
120 .observeOn(AndroidSchedulers.mainThread())
121 );
122 }
123
124 Flowable<MenuUiState> combinedRowStates = Flowable.combineLatest(result.rowUiStates.values(), (Object[] rowUiStates) -> {
125 int damage = 0;
126 boolean reset = false;
127 boolean weather = false;
128 for (Object state : rowUiStates) {
129 RowUiState rowUiState = (RowUiState) state;
130 damage += rowUiState.getDamage();
131 reset |= rowUiState.isHorn() || rowUiState.getUnits() != 0;
132 weather |= rowUiState.isWeather();
133 }
134 reset |= weather;
135 return new MenuUiState(damage, reset, weather, false);
136 }).distinctUntilChanged().onBackpressureLatest();
137
138 result.menuUiState = Flowable.combineLatest(combinedRowStates,
139 repository.hasNonEpicUnitsFlowable(),
140 (menuUiState, hasNonEpicUnits) -> new MenuUiState(menuUiState.getDamage(),
141 menuUiState.isReset(),
142 menuUiState.isWeather(),
143 hasNonEpicUnits))
144 .distinctUntilChanged()
145 .onBackpressureLatest()
146 .debounce(10, TimeUnit.MILLISECONDS)
147 .subscribeOn(Schedulers.io())
148 .observeOn(AndroidSchedulers.mainThread());
149
150 result.soundManager = soundManager;
151 return result;
152 }
