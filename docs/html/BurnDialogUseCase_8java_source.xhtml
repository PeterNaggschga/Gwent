37 private static Single<List<UnitEntity>> getBurnUnits(@NonNull UnitRepository repository) {
38 return repository.getUnits()
39 .flatMap(units -> {
40 if (units.isEmpty()) {
41 return Single.just(units);

43
44 Single<Map<RowType, DamageCalculator>> calculators = Single.just(new HashMap<>(RowType.values().length));
45 for (RowType row : RowType.values()) {
46 calculators = calculators.zipWith(DamageCalculatorUseCase.getDamageCalculator(repository, row), (calculatorMap, damageCalculator) -> {
47 calculatorMap.put(row, damageCalculator);
48 return calculatorMap;
49 });
50 }
51
52 return calculators.map(damageCalculators -> {
53 List<UnitEntity> maxDamageUnits = new ArrayList<>(units.size());
54 final int[] maxDamage = {0};
55 units.stream()
56 .filter(unit -> !unit.isEpic())
57 .forEach(unit -> {
58 int damage = unit.calculateDamage(Objects.requireNonNull(damageCalculators.get(unit.getRow())));
59 if (damage > maxDamage[0]) {
60 maxDamage[0] = damage;
61 maxDamageUnits.clear();
62 maxDamageUnits.add(unit);
63 } else if (damage == maxDamage[0]) {
64 maxDamageUnits.add(unit);
65 }
66 });
67 return maxDamageUnits;
68 });
69 });
