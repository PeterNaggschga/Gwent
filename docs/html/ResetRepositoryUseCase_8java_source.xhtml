45 @NonNull
46 static Maybe<UnitEntity> reset(@NonNull Context context, @NonNull UnitRepository repository,
47 boolean keepUnit, @NonNull SoundManager soundManager) {
48 return repository.getUnits()
49 .flatMapMaybe(units -> {
50 Optional<UnitEntity> keptUnit = keepUnit ? getRandomUnit(units) : Optional.empty();
51 long revengeUnits = units.stream()
52 .filter(unit -> unit.getAbility() == Ability.REVENGE)
53 .count() - (keptUnit.isPresent() && keptUnit.get().getAbility() == Ability.REVENGE ? 1 : 0);
54 Completable resultAction = (revengeUnits == 0) ?
55 repository.reset(keptUnit.orElse(null)) :
56 Completable.create(emitter ->
57 getRevengeDialog(context, repository, emitter, keptUnit.orElse(null), (int) revengeUnits, soundManager).show()
58 );
59 return resultAction.andThen(Maybe.fromOptional(keptUnit));
60 });
61 }

71 private static Optional<UnitEntity> getRandomUnit(@NonNull List<UnitEntity> units) {
72 units = units.stream()
73 .filter(unit -> !unit.isEpic())
74 .collect(Collectors.toList());
75 return units.isEmpty() ? Optional.empty() : Optional.of(units.get(new Random().nextInt(units.size())));
