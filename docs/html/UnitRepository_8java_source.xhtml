118 @NonNull
119 private Completable insertUnit(boolean epic, @IntRange(from = 0, to = UnitEntity.NON_EPIC_DAMAGE_VALUES_UPPER_BOUND) int damage, @NonNull Ability ability,
120 @IntRange(from = 1) @Nullable Integer squad, @NonNull RowType row) {
121 if (damage < 0 || damage > UnitEntity.NON_EPIC_DAMAGE_VALUES_UPPER_BOUND) {
122 throw new IllegalArgumentException("Damage be between 0 and " + UnitEntity.NON_EPIC_DAMAGE_VALUES_UPPER_BOUND + " but is " + damage + ".");
123 }
124 if (epic && Arrays.stream(UnitEntity.EPIC_DAMAGE_VALUES).noneMatch(integer -> integer == damage)) {
125 throw new IllegalArgumentException("Damage of epic units must be in " + Arrays.toString(UnitEntity.EPIC_DAMAGE_VALUES) + " but is " + damage + ".");
126 }
127 if (ability != Ability.BINDING && squad != null) {
128 throw new IllegalArgumentException("Squad must be null or ability must be BINDING but squad is " + squad + " and ability is " + ability + ".");
129 }
130 if (ability == Ability.BINDING && (squad == null || squad < 1)) {
131 throw new IllegalArgumentException("Squad must not be null or less than one but squad is " + squad + ".");
132 }
133 return database.units().insertUnit(epic, damage, ability, squad, row)
134 .subscribeOn(Schedulers.io())
135 .observeOn(AndroidSchedulers.mainThread());
136 }

337 @NonNull
338 public Flowable<List<UnitEntity>> getUnitsFlowable(@NonNull RowType row) {
339 return database.units()
340 .getUnitsFlowable(row)
341 .onBackpressureLatest()
342 .debounce(10, TimeUnit.MILLISECONDS)
343 .distinctUntilChanged((list1, list2) -> {
344 if (list1.size() != list2.size()) {
345 return false;
346 }
347 Iterator<UnitEntity> list1Iterator = list1.iterator();
348 return list2.stream().map(UnitEntity::getId).allMatch(id -> id.equals(list1Iterator.next().getId()));
349 })
350 .subscribeOn(Schedulers.io())
351 .observeOn(AndroidSchedulers.mainThread());
352 }
