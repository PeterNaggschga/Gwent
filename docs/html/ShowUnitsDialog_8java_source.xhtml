93 @NonNull
94 public static Single<ShowUnitsDialog> getDialog(@NonNull Context context, @NonNull RowType row, @NonNull SoundManager soundManager) {
95 return GwentApplication.getRepository(context)
96 .flatMap(repository -> repository.isWeather(row)
97 .zipWith(repository.isHorn(row), (weather, horn) ->
98 new CardUiStateFactory(context, weather, horn))
99 .map(factory -> {
100 CompositeDisposable initialDisposables = new CompositeDisposable();
101 CardListAdapter adapter = new CardListAdapter(
102 id -> initialDisposables.add(
103 repository
104 .copy(id)
105 .andThen(repository.getUnit(id))
106 .map(UnitEntity::isEpic)
107 .subscribe(epic -> soundManager.playCardAddSound(row, epic))
108 ),
109 id -> initialDisposables.add(
110 RemoveUnitsUseCase.remove(context, repository, id, soundManager)
111 .subscribe(soundManager::playCardRemovedSound,
112 throwable -> Log.e(ShowUnitsDialog.class.getSimpleName(),
113 "There has been an error with the removal of a unit. " +
114 "A reason might be tapping delete buttons too fast!",
115 throwable))
116 )
117 );
118 initialDisposables.add(
119 repository.getUnitsFlowable(row)
120 .map(factory::createCardUiState)
121 .subscribe(adapter::submitList)
122 );
123 ShowUnitsDialog result = new ShowUnitsDialog(context, row, adapter, soundManager);
124 result.disposables.add(initialDisposables);
125 return result;
126 })
127 );
128 }
