89 public CardUiState createCardUiState(@NonNull UnitEntity unit, @NonNull DamageCalculator calculator) {
90 int damageBackgroundId = R.drawable.icon_damage_background;
91 int damage = unit.calculateDamage(calculator);
92
93 if (unit.isEpic()) {

98 case 8:
99 damageBackgroundId = R.drawable.icon_epic_damage_8;

101 case 10:

103 break;
104 case 11:
105 damageBackgroundId = R.drawable.icon_epic_damage_11;
106 break;
107 case 15:
108 damageBackgroundId = R.drawable.icon_epic_damage_15;
109 break;
110 case 0:
111 default:
112 damageBackgroundId = R.drawable.icon_epic_damage_0;
113 }
114 damage = CardUiState.UNUSED;
115 }
116
117 int damageColor = Objects.requireNonNull(damageTextColors.get(unit.isBuffed(calculator)));
118
119 int abilityImage;
120 switch (unit.getAbility()) {
121 case HORN:
122 abilityImage = R.drawable.icon_horn;
123 break;
124 case REVENGE:
125 abilityImage = R.drawable.icon_revenge;
126 break;
127 case BINDING:
128 abilityImage = R.drawable.icon_binding;
129 break;
130 case MORAL_BOOST:
131 abilityImage = R.drawable.icon_moral_boost;
132 break;
133 case NONE:
134 default:
135 abilityImage = CardUiState.UNUSED;
136 }
137
138 return new CardUiState(unit.getId(), damageBackgroundId, damage, damageColor, abilityImage, unit.getSquad());
