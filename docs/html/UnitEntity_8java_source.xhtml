165 @NonNull
166 public static String collectionToString(@NonNull Context context, @NonNull Collection<UnitEntity> units) {
167 if (units.isEmpty()) {
168 throw new IllegalArgumentException("Units collection must not be empty.");
169 }
170
171 Map<String, Integer> descriptionStrings = new HashMap<>(units.size());
172 for (UnitEntity unit : units) {
173 String description = unit.toString(context);
174 descriptionStrings.merge(description, 1, Integer::sum);
175 }
176
177 List<String> descriptions = new ArrayList<>(descriptionStrings.size());
178 if (descriptionStrings.size() < units.size()) {
179 descriptionStrings.forEach((key, value) ->
180 descriptions.add(context.getString(R.string.unit_toString_multiplicity, value, key)));
181 } else {
182 descriptions.addAll(descriptionStrings.keySet());
183 }
184
185 Iterator<String> descriptionIterator = descriptions.iterator();
186 String[] result = {descriptionIterator.next()};
187 if (descriptionIterator.hasNext()) {
188 result[0] = context.getString(R.string.unit_collection_toString_accumulation_word,
189 descriptionIterator.next(),
190 result[0])
191 .trim();
192 }
193 descriptionIterator.forEachRemaining(description ->
194 result[0] = context.getString(R.string.unit_collection_toString_accumulation_symbol,
195 description,
196 result[0])
197 .trim());
198 return result[0];
199 }
