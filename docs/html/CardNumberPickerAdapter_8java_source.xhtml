152 static NumberPicker.OnValueChangeListener getDelayedOnValueChangeListener(@NonNull NumberPicker.OnValueChangeListener originalListener) {
153 return (picker, oldVal, newVal) -> new Handler(Looper.getMainLooper()).postDelayed(() -> {
154 if (newVal == picker.getValue()) {

165 @NonNull
166 Single<Boolean> addSelectedUnits(@NonNull RowType row) {
167 // handle delayed events due to delayed OnValueChangedListeners
168 // delayed changes in damagePicker after hero selection
169 int damage;
170 if (epicPicker.getValue()) {
171 damage = Arrays.stream(UnitEntity.EPIC_DAMAGE_VALUES).anyMatch(integer -> Objects.equals(integer, damagePicker.getValue())) ? damagePicker.getValue() : UnitEntity.EPIC_DAMAGE_VALUES[3];
172 } else {
173 damage = damagePicker.getValue();
174 }
175
176 // delayed changes in squadPicker after squad selection
177 Integer squad;
178 if (abilityPicker.getValue() == Ability.BINDING) {
179 squad = squadPicker.getVisibility() == View.VISIBLE ? squadPicker.getValue() : squadManager.getFirstSquadWithMembers();
180 } else {
181 squad = null;
182 }
183 return GwentApplication.getRepository(numberPicker.getContext())
184 .flatMapCompletable(repository ->
185 repository.insertUnit(epicPicker.getValue(),
186 damage,
187 abilityPicker.getValue(),
188 squad,
189 row,
190 numberPicker.getValue()))
191 .andThen(Single.just(epicPicker.getValue()));
192 }
