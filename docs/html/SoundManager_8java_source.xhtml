181 SharedPreferences pref = PreferenceManager.getDefaultSharedPreferences(context);
182 changeListener = (sharedPreferences, key) -> {
183 if (context.getResources().getString(R.string.preference_key_sounds_all).equals(key)) {
184 if (sharedPreferences.getBoolean(key,
185 context.getResources().getBoolean(R.bool.sound_preference_default))) {
186 Arrays.stream(sounds).forEach(sound -> sound.setActivated(sharedPreferences));
187 } else {
188 Arrays.stream(sounds).forEach(sound -> sound.setActivated(false));
189 }
190 } else {
191 Arrays.stream(sounds)
192 .filter(sound -> sound.getPreferenceKey().equals(key))
193 .forEach(sound -> sound.setActivated(sharedPreferences.getBoolean(key,
194 context.getResources().getBoolean(R.bool.sound_preference_default))));
195 }
196 };
197 pref.registerOnSharedPreferenceChangeListener(changeListener);
